create database tecinv;
use tecinv;

create table usuario(
  id int AUTO_INCREMENT not null,
  nome varchar(50) not null,
  email varchar(50) unique not null,
  senha varchar(100) not null,
  PRIMARY key (id)
);

create table categoria_equipamento(
  id int AUTO_INCREMENT not null,
  nome_categoria varchar(50) unique not null,
  PRIMARY key (id)
 );
 
CREATE table secretaria(
  id int AUTO_INCREMENT not null,
  nome_secretaria varchar(100) unique not null,
  PRIMARY key (id)
  );
 
 CREATE table equipamento_tecnologico(
   id int AUTO_INCREMENT not null,
   usuario_id int not null,
   categoria_id int not null,
   secretaria_id int not null,
   fabricante varchar(100) not null,
   modelo varchar(100) not null,
   num_patrimonio varchar(100) unique not null,
   responsavel varchar(100) not null,
   setor varchar(100) not null,
   especificacoes text not null,
   observacoes text null,
   PRIMARY key (id),
   FOREIGN key (usuario_id) REFERENCES usuario(id),
   FOREIGN key (categoria_id) REFERENCES categoria_equipamento(id),
   FOREIGN key (secretaria_id) REFERENCES secretaria(id)
   );
